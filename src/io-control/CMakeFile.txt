# CMakeLists.txt
cmake_minimum_required(VERSION 3.12)
project(gpio_sequence C)

set(CMAKE_C_STANDARD 11)
set(CMAKE_C_STANDARD_REQUIRED ON)

find_package(PkgConfig REQUIRED)
pkg_check_modules(LIBGPIOD REQUIRED libgpiod)
find_package(Threads REQUIRED)

add_executable(gpio_sequence
    main.c
)

target_include_directories(gpio_sequence PRIVATE
    ${LIBGPIOD_INCLUDE_DIRS}
)

target_link_libraries(gpio_sequence
    ${LIBGPIOD_LIBRARIES}
    Threads::Threads
)

install(TARGETS gpio_sequence DESTINATION bin)

if(NOT LIBGPIOD_FOUND)
    message(FATAL_ERROR "需要安装libgpiod: sudo apt install libgpiod-dev")
endif()
